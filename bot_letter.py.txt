import os
import logging
from telegram import Update
from telegram.ext import ApplicationBuilder, ContextTypes, CommandHandler, MessageHandler, filters

# 1. DATOS: Tus listas de transformación
letras_normales = "abcdefghijklmnopqrstuvwxyz"
letras_font =     ['?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?', '?']

# 2. LÓGICA: El motor que usa .index()
def transformar_texto(texto):
    resultado = ""
    for caracter in texto():
        if caracter in letras_normales:
            posicion = letras_normales.index(caracter)
            resultado += letras_font[posicion]
        else:
            resultado += caracter
    return resultado

# 3. FUNCIONES DEL BOT
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("ˇHola! Envíame cualquier texto y lo transformaré.")

async def procesar_mensaje(update: Update, context: ContextTypes.DEFAULT_TYPE):
    texto_usuario = update.message.text
    texto_final = transformar_texto(texto_usuario)
    await update.message.reply_text(texto_final)

# 4. ARRANQUE
if __name__ == '__main__':
    # Esta línea lee el token desde la configuración de Render (más seguro)
    TOKEN = os.environ.get("TELEGRAM_TOKEN")
    
    # Si no hay token en el servidor, usamos uno de prueba (opcional)
    if not TOKEN:
        TOKEN = "TU_TOKEN_AQUI_PARA_PROBAR_EN_LOCAL"

    app = ApplicationBuilder().token(TOKEN).build()
    
    app.add_handler(CommandHandler("start", start))
    app.add_handler(MessageHandler(filters.TEXT & (~filters.COMMAND), procesar_mensaje))
    
    print("Bot encendido y listo...")
    app.run_polling()
